```python
def calculate_detailed_sale_cost(sale_price: float) -> float:
    """詳細な売却コスト計算（仲介手数料+その他）"""
    # 仲介手数料（段階的計算）
    if sale_price <= 200:  # 200万円以下
        commission = sale_price * 0.05
    elif sale_price <= 400:  # 400万円以下
        commission = 200 * 0.05 + (sale_price - 200) * 0.04
    else:  # 400万円超
        commission = 200 * 0.05 + 200 * 0.04 + (sale_price - 400) * 0.03
    
    # その他費用（登記費用、印紙代等：1-2%）
    other_costs = sale_price * 0.015
    
    return commission + other_costs

# 277行目を修正

```
